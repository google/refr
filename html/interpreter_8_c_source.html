00046 void
00047 Interpreter::Eval(StreamTokenizer &st) {
00048   // Keeps reading assignment statements until there are no more tokens.
00049   while (st.PeekTokenType() != StreamTokenizer::EOF_TYPE) {
00050     // Read variable name.
00051     StreamTokenizer::TokenType token_type = st.PeekTokenType();
00052     if (token_type != StreamTokenizer::IDENTIFIER) {
00053       WrongTokenTypeError(st.PeekTokenStart(), StreamTokenizer::IDENTIFIER,
00054                           token_type, st.Peek());
00055     }
00056     string varname = st.Next();
00057 
00058     // Next, read equals sign.
00059     token_type = st.PeekTokenType();
00060     if (st.Peek() != "=") {
00061       WrongTokenError(st.PeekTokenStart(), "=", st.Peek(), st.PeekTokenType());
00062     }
00063     /*
00064     if (token_type != StreamTokenizer::RESERVED_CHAR) {
00065       WrongTokenTypeError(st.PeekTokenStart(), StreamTokenizer::RESERVED_CHAR,
00066                           token_type, st.Peek());
00067     }
00068     */
00069 
00070     // Consume equals sign.
00071     st.Next();
00072 
00073     if (st.PeekTokenType() == StreamTokenizer::EOF_TYPE) {
00074       ostringstream err_ss;
00075       err_ss << "Interpreter:" << filename_
00076              << ": error: unexpected EOF at stream position "
00077              << st.tellg();
00078       throw std::runtime_error(err_ss.str());
00079     }
00080 
00081     // Consume and set the value for this variable in the environment.
00082     env_->ReadAndSet(varname, st);
00083 
00084     token_type = st.PeekTokenType();
00085     if (st.Peek() != ";") {
00086       WrongTokenError(st.PeekTokenStart(), ";", st.Peek(), st.PeekTokenType());
00087     }
00088     // Consume semicolon.
00089     st.Next();
00090   }
00091 }
